  <div class="container">
    <Tooltip
      content="提示文字"
      placement="left"
    >
      <Button>左 - left</Button>
    </Tooltip>
    <Tooltip
      content="提示文字"
      placement="top"
    >
      <Button>上 - top</Button>
    </Tooltip>
    <Tooltip
      content="提示文字"
      placement="bottom"
    >
      <Button>下 - bottom</Button>
    </Tooltip>
    <Tooltip
      content="提示文字"
      placement="right"
    >
      <Button>右 - right</Button>
    </Tooltip>
  </div>
</template>

<script>
  import {Tooltip, Button} from "@wanmaoor/giaoui"
  export default {
  	name: "Tooltips",
  	components: {Button, Tooltip}
  }
</script>
